\begin{frame}[fragile]

  \begin{code}
type alias Model =
  List (Char, Bool)
  \end{code}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}[fragile]

  \begin{code}
init : String -> Model
init word =
  String.foldr (::) [] word
    |> List.map (\letter -> (letter, False))
  \end{code}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}[fragile]

  \begin{code}
update : Action -> Model -> Model
update guess letters =
  let
    match guess (letter, guessed) =
      (letter, guess == letter || guessed)
  in
    List.map (match guess) letters
  \end{code}
\end{frame}
